    CREATE TABLE Producto(

        Cantidad_producto INT,
        Precio_Producto FLOAT,

    CREATE TABLE Empleado(

        id_proveedor INT PRIMARY KEY AUTO_INCREMENT,
        Nombre_proveedor VARCHAR(50),

        FOREIGN KEY (id_producto) REFERENCES Producto(id)

    CREATE TABLE Venta(

        FOREIGN KEY (id_Cliente) REFERENCES Cliente(id_Cliente),

        fecha_registro TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

-- Tabla de usuarios del sistema
-- La ligamos a Empleado porque cada cajero/gerente YA es un empleado
CREATE TABLE IF NOT EXISTS Usuario (
    id_usuario    INT AUTO_INCREMENT PRIMARY KEY,
    id_empleado   INT NOT NULL UNIQUE,
    username      VARCHAR(60) NOT NULL UNIQUE,
    password_hash VARCHAR(255) NOT NULL,
    rol           ENUM('gerente','cajero','almacenista') NOT NULL DEFAULT 'cajero',
    activo        TINYINT(1) NOT NULL DEFAULT 1,
    FOREIGN KEY (id_empleado) REFERENCES Empleado(Id_Empleado) ON DELETE CASCADE
);

-- ── 1. CONFIGURACION_TIENDA ───────────────────────────────────
-- Guarda los datos del negocio: nombre, RFC, dirección, etc.
-- Solo hay un registro (id=1). Se lee en tickets y reportes.
-- Usamos clave-valor (clave VARCHAR, valor TEXT) para que sea
-- flexible: puedes agregar cualquier configuración sin alterar
-- la estructura de la tabla.
CREATE TABLE IF NOT EXISTS Configuracion_Tienda (
    id_config   INT AUTO_INCREMENT PRIMARY KEY,
    clave       VARCHAR(60)  NOT NULL UNIQUE,  -- ej: "nombre_tienda", "rfc"
    valor       TEXT,                          -- ej: "Abarrotes Don Pepe", "XAXX010101000"
    descripcion VARCHAR(120)                   -- para que el admin sepa qué es cada clave
);

    id_caja      INT AUTO_INCREMENT PRIMARY KEY,
    nombre       VARCHAR(60) NOT NULL DEFAULT 'Caja Principal',
    descripcion  TEXT,
    activa       TINYINT(1) NOT NULL DEFAULT 1
);

-- ── 3. CORTE_CAJA ─────────────────────────────────────────────
-- Cada vez que un cajero abre su turno se crea un registro aquí.
-- Cuando cierra el turno se llena fecha_cierre y fondo_cierre.
--
-- ¿Por qué es importante?
-- Permite saber exactamente cuánto dinero entró en el turno de
-- cada empleado y si cuadra con las ventas registradas.
CREATE TABLE IF NOT EXISTS Corte_Caja (
    id_corte      INT AUTO_INCREMENT PRIMARY KEY,
    id_caja       INT NOT NULL,
    id_usuario    INT NOT NULL,              -- quién abrió la caja
    fecha_apertura DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    fondo_apertura DECIMAL(10,2) NOT NULL DEFAULT 0.00,  -- dinero inicial en caja
    fecha_cierre   DATETIME,                -- NULL mientras el turno está abierto
    fondo_cierre   DECIMAL(10,2),           -- dinero contado al cerrar
    total_ventas   DECIMAL(10,2),           -- suma de ventas del turno (calculado)
    notas          TEXT,
    FOREIGN KEY (id_caja)    REFERENCES Caja(id_caja),
    FOREIGN KEY (id_usuario) REFERENCES Usuario(id_usuario)
);


-- ── 4. PAGO_VENTA ─────────────────────────────────────────────
-- Registra cómo se pagó cada venta: efectivo, tarjeta, transferencia...
-- Una venta puede tener múltiples pagos (pago mixto: parte efectivo + parte tarjeta).
--
-- ¿Por qué separada de Venta?
-- La tabla Venta ya existe y no queremos modificarla (regla del PBI-004:
-- no romper tablas existentes). Agregamos esta tabla relacionada.
CREATE TABLE IF NOT EXISTS Pago_Venta (
    id_pago        INT AUTO_INCREMENT PRIMARY KEY,
    id_venta       INT NOT NULL,
    forma_pago     ENUM('efectivo','tarjeta','transferencia','otro')
                    NOT NULL DEFAULT 'efectivo',
    monto          DECIMAL(10,2) NOT NULL,
    referencia     VARCHAR(100),            -- número de autorización de tarjeta, etc.
    fecha_pago     DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_venta) REFERENCES Venta(id_Venta) ON DELETE CASCADE
    -- ON DELETE CASCADE: si se elimina la venta, se eliminan sus pagos también
);


-- ── 5. AUDITORIA ──────────────────────────────────────────────
-- Bitácora de operaciones importantes: quién creó, editó o eliminó qué.
-- Es de solo inserción — nunca se modifica ni elimina un registro de auditoría.
--
-- ¿Cómo funciona?
-- Cuando el backend ejecuta una operación importante (crear venta, eliminar
-- producto, cambiar precio), inserta un registro aquí. Así el gerente puede
-- ver el historial completo de cambios.
CREATE TABLE IF NOT EXISTS Auditoria (
    id_auditoria  INT AUTO_INCREMENT PRIMARY KEY,
    id_usuario    INT,                      -- quién hizo la acción (NULL si fue el sistema)
    tabla_afectada VARCHAR(60) NOT NULL,    -- ej: "Producto", "Venta"
    operacion     ENUM('INSERT','UPDATE','DELETE') NOT NULL,
    id_registro   INT,                      -- el id del registro afectado
    detalle       TEXT,                     -- descripción legible: "Cambió precio de $10 a $12"
    fecha         DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_usuario) REFERENCES Usuario(id_usuario) ON DELETE SET NULL
    -- ON DELETE SET NULL: si se borra el usuario, la auditoría queda con NULL
    -- (preferible a perder el historial)
);
